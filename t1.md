```python
import torch
import torch.distributed as dist
import torch.nn.functional as F
from torch import nn

# ======================== Config ========================

NUM_EXPERTS = 8
TOP_K = 2
DIM = 2048
HIDDEN_DIM = 512
NUM_TOKENS = 4
SCORE_FUNC = "sigmoid"


def setup():
    dist.init_process_group("nccl")
    rank = dist.get_rank()
    world_size = dist.get_world_size()
    torch.cuda.set_device(rank)
    return rank, world_size


def log(msg):
    print(f"[Rank {torch.distributed.get_rank()}] {msg}", flush=True)


# ======================== Step 1: Router ========================

class Router(nn.Module):
    """
    Token-choice top-k router.
    Input:  x (T, D)
    Output: top_scores (T, top_k), selected_experts (T, top_k)
    """

    def __init__(self, dim, num_experts, top_k, score_func="sigmoid"):
        super().__init__()
        self.gate = nn.Linear(dim, num_experts, bias=False)
        self.num_experts = num_experts
        self.top_k = top_k
        self.score_func = score_func

    def forward(self, x):
        logits = self.gate(x)  # (T, num_experts)
        if self.score_func == "sigmoid":
            scores = torch.sigmoid(logits.float())
        else:
            scores = F.softmax(logits.float(), dim=-1)

        top_scores, selected_experts = torch.topk(scores, k=self.top_k, dim=-1)
        return top_scores, selected_experts


# ======================== Step 2: Reorder #1 ========================

def reorder_by_expert(selected_experts, num_experts):
    """Sort tokens by global expert id so that tokens for the same expert are contiguous."""
    flat_experts = selected_experts.view(-1)
    token_indices_sorted = torch.argsort(flat_experts, stable=True)
    num_tokens_per_expert = torch.histc(
        flat_experts.float(), bins=num_experts, min=0, max=num_experts,
    )
    return token_indices_sorted, num_tokens_per_expert


# ======================== Step 3: All-to-All Dispatch ========================

def dispatch(routed_input, num_tokens_per_expert, world_size):
    """
    Dispatch tokens to the rank owning each expert via all-to-all.

    Steps:
      a) Exchange per-expert counts so each rank knows how many tokens it will receive.
      b) Compute input_splits / output_splits from the counts.
      c) All-to-all to send token data.

    After dispatch, the received data on each rank is grouped by source rank:
        [E0_from_r0, E1_from_r0, E0_from_r1, E1_from_r1, ...]
    """
    num_local_experts = NUM_EXPERTS // world_size

    # a) Exchange counts
    recv_counts = torch.zeros_like(num_tokens_per_expert)
    dist.all_to_all_single(recv_counts, num_tokens_per_expert)

    log(f"num_tokens_per_expert: {num_tokens_per_expert} -> recv_counts: {recv_counts}")

    # b) Compute splits
    # input_splits[i] = total tokens this rank sends to rank i
    input_splits = (
        num_tokens_per_expert.view(world_size, num_local_experts)
        .sum(dim=1).int().tolist()
    )
    # output_splits[i] = total tokens this rank receives from rank i
    output_splits = (
        recv_counts.view(world_size, num_local_experts)
        .sum(dim=1).int().tolist()
    )

    # c) All-to-all token data
    recv_buffer = torch.zeros(
        sum(output_splits), DIM,
        device=routed_input.device, dtype=routed_input.dtype,
    )
    dist.all_to_all_single(recv_buffer, routed_input, output_splits, input_splits)

    log(f"  dispatch: {routed_input.shape} -> {recv_buffer.shape}")
    return recv_buffer, input_splits, output_splits, recv_counts


# ======================== Step 4: Reorder #2 (permute) ========================

def permute_for_experts(recv_tokens, recv_counts, world_size, num_local_experts):
    """
    After all-to-all, tokens are grouped by source rank:
        [E0_from_r0, E1_from_r0, E0_from_r1, E1_from_r1]

    Permute to group by local expert for computation:
        [E0_from_r0, E0_from_r1, E1_from_r0, E1_from_r1]
        i.e., [all_E0, all_E1]

    Returns:
        permuted_tokens, tokens_per_local_expert, forward_indices (for unpermute)
    """
    counts = recv_counts.view(world_size, num_local_experts).int()
    chunk_sizes = counts.view(-1).tolist()
    chunks = recv_tokens.split(chunk_sizes, dim=0)

    # Reorder chunks: iterate expert-first, then rank
    permuted_chunks = []
    tokens_per_local_expert = []
    for ej in range(num_local_experts):
        total = 0
        for ri in range(world_size):
            permuted_chunks.append(chunks[ri * num_local_experts + ej])
            total += chunk_sizes[ri * num_local_experts + ej]
        tokens_per_local_expert.append(total)
    permuted_tokens = torch.cat(permuted_chunks, dim=0)

    # Build forward_indices for unpermute: forward_indices[i] = original position of permuted[i]
    offsets = []
    off = 0
    for s in chunk_sizes:
        offsets.append(off)
        off += s

    fwd = []
    for ej in range(num_local_experts):
        for ri in range(world_size):
            idx = ri * num_local_experts + ej
            fwd.extend(range(offsets[idx], offsets[idx] + chunk_sizes[idx]))
    forward_indices = torch.tensor(fwd, device=recv_tokens.device)

    return permuted_tokens, tokens_per_local_expert, forward_indices


# ======================== Step 5: Expert Compute ========================

class Experts(nn.Module):
    """Gated FFN experts (SwiGLU). Each expert: SiLU(x @ W1^T) * (x @ W3^T) @ W2^T"""

    def __init__(self, num_experts, dim, hidden_dim):
        super().__init__()
        self.w1 = nn.Parameter(torch.randn(num_experts, hidden_dim, dim) * 0.02)
        self.w2 = nn.Parameter(torch.randn(num_experts, dim, hidden_dim) * 0.02)
        self.w3 = nn.Parameter(torch.randn(num_experts, hidden_dim, dim) * 0.02)

    def forward(self, x, tokens_per_expert):
        """x: (total_tokens, dim), tokens_per_expert: list[int]"""
        splits = torch.split(x, tokens_per_expert, dim=0)
        outs = []
        for i, xi in enumerate(splits):
            if xi.shape[0] == 0:
                continue
            
            h = F.silu(xi @ self.w1[i].T) * (xi @ self.w3[i].T)
            log(f"xi: {h.shape} {self.w2[i].T.shape}")
            outs.append(h @ self.w2[i].T)
        return torch.cat(outs, dim=0)


# ======================== Step 6: Un-reorder #2 (unpermute) ========================

def unpermute(expert_output, forward_indices):
    """Reverse the expert-grouped permutation back to rank-grouped layout."""
    out = torch.empty_like(expert_output)
    out[forward_indices] = expert_output
    return out


# ======================== Step 7: All-to-All Combine ========================

def combine(expert_output, input_splits, output_splits):
    """All-to-all to send expert results back to the originating ranks (reverse of dispatch)."""
    combined = torch.zeros(
        sum(input_splits), DIM,
        device=expert_output.device, dtype=expert_output.dtype,
    )
    dist.all_to_all_single(combined, expert_output, input_splits, output_splits)
    log(f"  combine: {expert_output.shape} -> {combined.shape}")
    return combined


# ======================== Step 8: Un-reorder #1 + Weighted Sum ========================

def unreorder_and_aggregate(combined_output, token_indices_sorted, top_scores, num_tokens, top_k):
    """Restore original token order and compute weighted sum over top-k expert outputs."""
    dim = combined_output.shape[-1]

    # Scatter back to original order
    output = torch.zeros_like(combined_output)
    output[token_indices_sorted] = combined_output

    # Weighted sum (score_after_experts): (T, 1, top_k) @ (T, top_k, D) -> (T, D)
    output = output.reshape(num_tokens, top_k, dim)
    output = torch.bmm(
        top_scores.unsqueeze(1),
        output.float(),
    ).squeeze(1).to(combined_output.dtype)

    return output


# ======================== Main ========================

def main():
    rank, world_size = setup()
    device = torch.device(f"cuda:{rank}")

    assert NUM_EXPERTS % world_size == 0, \
        f"NUM_EXPERTS ({NUM_EXPERTS}) must be divisible by world_size ({world_size})"
    num_local_experts = NUM_EXPERTS // world_size

    log("=" * 60)
    log(f"world_size={world_size}, num_experts={NUM_EXPERTS}, "
        f"local_experts={num_local_experts}, top_k={TOP_K}")
    log(f"T={NUM_TOKENS}, D={DIM}, hidden={HIDDEN_DIM}")
    log("=" * 60)

    torch.manual_seed(42 + rank)
    x = torch.randn(NUM_TOKENS, DIM, device=device)  # (T, D)

    # Step 1: Router
    log(f"\n{'='*20} Step 1: Router {'='*20}")
    router = Router(DIM, NUM_EXPERTS, TOP_K, SCORE_FUNC).to(device)
    top_scores, selected_experts = router(x)
    log(f"  selected_experts: {selected_experts}")
    log(f"  top_scores: {top_scores.shape}, selected_experts: {selected_experts.shape}")

    # Step 2: Reorder #1 - sort by global expert id
    log(f"\n{'='*20} Step 2: Reorder #1 {'='*20}")
    token_indices_sorted, num_tokens_per_expert = reorder_by_expert(
        selected_experts, NUM_EXPERTS,
    )
    routed_input = x[token_indices_sorted // TOP_K]  # (T*top_k, D)
    log(f"  routed_input: {routed_input.shape}")

    # Step 3: All-to-All Dispatch
    log(f"\n{'='*20} Step 3: Dispatch {'='*20}")
    recv_tokens, input_splits, output_splits, recv_counts = dispatch(
        routed_input, num_tokens_per_expert, world_size,
    )

    # Step 4: Reorder #2 - permute from rank-grouped to expert-grouped
    log(f"\n{'='*20} Step 4: Permute {'='*20}")
    permuted_tokens, tokens_per_expert, fwd_indices = permute_for_experts(
        recv_tokens, recv_counts, world_size, num_local_experts,
    )
    log(f"  tokens_per_local_expert: {tokens_per_expert} {permuted_tokens.shape}")

    # Step 5: Expert Compute
    log(f"\n{'='*20} Step 5: Experts {'='*20}")
    experts = Experts(num_local_experts, DIM, HIDDEN_DIM).to(device)
    expert_output = experts(permuted_tokens, tokens_per_expert)
    log(f"  expert_output: {expert_output.shape}")

    # Step 6: Un-reorder #2 - unpermute back to rank-grouped
    log(f"\n{'='*20} Step 6: Unpermute {'='*20}")
    unpermuted_output = unpermute(expert_output, fwd_indices)

    # Step 7: All-to-All Combine
    log(f"\n{'='*20} Step 7: Combine {'='*20}")
    combined_output = combine(unpermuted_output, input_splits, output_splits)

    # Step 8: Un-reorder #1 + Weighted Sum
    log(f"\n{'='*20} Step 8: Aggregate {'='*20}")
    output = unreorder_and_aggregate(
        combined_output, token_indices_sorted, top_scores, NUM_TOKENS, TOP_K,
    )
    log(f"  output: {output.shape}  (expected [{NUM_TOKENS}, {DIM}])")

    log("Done!")
    dist.destroy_process_group()
    


if __name__ == "__main__":
    main()
```
